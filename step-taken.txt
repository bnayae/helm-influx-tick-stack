# Update helm
helm init --upgrade

# create the helm scaffold
helm create influx-tick-stack

#requirements
requirements.yaml

# build requirements
# helm dependency build .\influx-tick-stack
helm dep update .\influx-tick-stack\

# dry-run
helm install --dry-run --debug ./influx-tick-stack

# dry-run with override
helm install --dry-run --debug ./influx-tick-stack --set service.internalPort=8080

# first deploy (override the node-type)
helm install --name tick-test ./influx-tick-stack --set service.type=NodePort

# get status helm status tick-test

# get pods
kubectl get pods

# get services
kubectl get services

# alternative (run the command from the notes.txt)
kubectl get --namespace default -o jsonpath="{.spec.ports[0].nodePort}" services tick-test-influx-tick-stack

# when deployed on remote cluster
kubectl get nodes --namespace default -o jsonpath="{.items[0].status.addresses[0].address}"

# browse to the service (use the port from [kubectl get services])
http://localhost:{port}/

# describe the service
kubectl describe services tick-test-influx-tick-stack

# upgrade
helm upgrade tick-test .\influx-tick-stack\

# view history
helm history tick-test

# rollback
helm rollback tick-test 1

# delete deployment
helm delete tick-test 
# delete permanently   
helm delete tick-test --purge
